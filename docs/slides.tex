\begin{frame}{Problem i motywacja}
    \begin{itemize}
        \item Dynamiczny rozwój dużych modeli językowych (LLM)
        \item Rosnąca złożoność IaC (m.in. Dockerfile, manifesty Kubernetes)
        \item IaC przyspiesza DevOps, ale stawia wysokie wymagania kompetencyjne
        \item Ręczne tworzenie Dockerfile i manifestów K8s jest czasochłonne i podatne na błędy
        \item LLM obiecują automatyzację, lecz brakuje twardej oceny end-to-end
        \item Kluczowy problem: czy możemy zaufać LLM w generowaniu infrastruktury produkcyjnej?
        \item Potrzebne są metryki skuteczności, jakości, powtarzalności i ryzyka
    \end{itemize}
    \vspace{0.5em}
    \textit{Generowanie kodu aplikacyjnego przez LLM jest dobrze zbadane, natomiast generowanie infrastruktury jako kodu -- znacznie słabiej.}
\end{frame}

\begin{frame}{Cel pracy i pytanie badawcze}
    \begin{itemize}
        \item Cel główny: ocena możliwości i ograniczeń w autonomicznej generacji konfiguracji IaC przez agenty LLM
        \item Zakres: pliki Dockerfile i manifesty Kubernetes
        \item Agenty LLM generujące konfiguracje IaC
        \item Brak ingerencji człowieka w generację - pełna autonomia
        \item Pytanie badawcze: Jak skuteczne, jakościowe, powtarzalne i bezpieczne są generowane konfiguracje IaC w różnych scenariuszach
    \end{itemize}
\end{frame}

\begin{frame}{Hipotezy badawcze (H1--H5)}
    \begin{itemize}
        \item H1: działające konfiguracje end-to-end
        \item H2: spadek skuteczności wraz ze złożonością
        \item H3: jakość konfiguracji i wpływ \textit{prompt engineering} na nią
        \item H4: powtarzalność procesu przy deterministycznych parametrach
        \item H5: podatność na manipulację kontekstem agenta (\textit{prompt injection})
    \end{itemize}
\end{frame}

\begin{frame}{Przegląd literatury}
    \begin{itemize}
        \item Generacja IaC: przegląd + Repo2Run \cite{srivatsa_survey_2024,hu_llm-based_2025}
        \item \textit{Prompt engineering}: ``Don't Train, Just Prompt'' + Ansible Wisdom \cite{kratzke_dont_2024,pujar_invited_2023}
        \item Walidacja / security: naprawa IaC + GenKubeSec \cite{low_repairing_2024,malul_genkubesec_2024}
        \item Konwersje Docker Compose $\rightarrow$ K8s: migracja + CloudEval-YAML \cite{ueno_migrating_2024,xu_cloudeval-yaml_2023}
    \end{itemize}
\end{frame}

\begin{frame}{Luki badawcze}
    \begin{itemize}
        \item Brak podejść autonomicznych (agent czyta kod źródłowy, nie tylko język naturalny)
        \item Brak badań nad granicą złożoności systemów dla IaC generowanego przez LLM
        \item Ograniczona obiektywna oceny jakości (build $\rightarrow$ apply $\rightarrow$ runtime)
        \item Brak systematycznej oceny powtarzalności procesów agentowych
        \item Ograniczona analiza bezpieczeństwa i manipulacji kontekstem (np. \textit{prompt injection})
    \end{itemize}
\end{frame}

\begin{frame}{Architektura systemu badawczego}
    \begin{columns}[T]
        \column{0.48\textwidth}
        \begin{itemize}
            \item Agent z użyciem narzędzi analizuje kod i generuje pliki konfiguracyjne
            \item Następnie uruchamia się potok walidacyjny: lint $\rightarrow$ build $\rightarrow$ apply
            \item Walidacja runtime: test dostępności aplikacji przez HTTP (ingress health check)
            \item MicroK8s zapewnia kontrolowane, powtarzalne środowisko, lokalny rejestr obrazów. Czyszczenie klastra przed każdą iteracją
            \item System zbiera metryki z każdego etapu eksperymentu
        \end{itemize}
        \column{0.52\textwidth}
        \centering
        \begin{tikzpicture}[
            node distance=0.6cm and 0.4cm,
            box/.style={draw, rounded corners, align=center, font=\footnotesize, minimum width=3.2cm, minimum height=0.7cm}
        ]
            \node[box] (repo) {Repozytorium};
            \node[box, below=of repo] (agent) {Agent LLM\\+ narzędzia};
            \node[box, below=of agent] (cfg) {Dockerfile\\+ manifesty K8s};
            \node[box, below=of cfg] (val) {Walidacja\\build/lint/apply/runtime};
            \node[box, below=of val] (out) {Metryki + raport};
            \draw[->] (repo) -- (agent);
            \draw[->] (agent) -- (cfg);
            \draw[->] (cfg) -- (val);
            \draw[->] (val) -- (out);
            \node[font=\footnotesize, right=0.2cm of val] {MicroK8s};
        \end{tikzpicture}
    \end{columns}
\end{frame}

\begin{frame}{Narzędzia agenta i prompt}
    \begin{itemize}
        \item Analiza repo: \texttt{clone\_repo}, \texttt{prepare\_repo\_tree}, \texttt{ls}
        \item Odczyt i wyszukiwanie: \texttt{get\_file\_content}, \texttt{search\_files}, \texttt{find\_files}
        \item Zapis i korekty: \texttt{write\_file}, \texttt{patch\_file}
        \item Pomocnicze: \texttt{base64\_encode}, \texttt{base64\_decode}, \texttt{think}
    \end{itemize}
\end{frame}

\begin{frame}{Prompt systemowy}
    \begin{itemize}
        \item Jedna próba, brak pętli sprzężenia zwrotnego
        \item Generacja Dockerfile i manifestów K8s dla wszystkich usług
        \item Wykrywanie zależności zewnętrznych (DB, cache, kolejki)
        \item Zasady K8s: Nazewnictwo hostów w Ingress, ustawianie requests/limits, securityContext, readiness/liveness probes
        \item Wyjście: JSON z listą obrazów i manifestów + \texttt{test\_endpoints}
        \item Co ważne: brak instrukcji specyficznych dla danych technologii lub "porad", mogących sugerować rozwiązania.
        \item Razem 6943 znaki. Pełna treść promptu w załączniku pracy magisterskiej.
    \end{itemize}
\end{frame}

\begin{frame}{Wykorzystane modele}
    \begin{itemize}
        \item OpenAI GPT-5 Mini
        \item Google Gemini 2.5 Flash
        \item DeepSeek V3.2
        \item Wybrano wiodące modele największych dostawców, w wariantach o mniejszej skali, ze względu na wysoki koszt inferencji, a niski budżet autora.
        \item Każdy przebieg (repozytorium + model + wariant prompta) uruchamiano co najmniej 2 razy.
    \end{itemize}
\end{frame}

\begin{frame}{Projekt H1 -- skuteczność}
    \begin{itemize}
        \item Cel: czy agent generuje działające konfiguracje IaC
        \item Zbiór: 25 publicznych repozytoriów webowych z GitHuba
        \item Wyszukiwanie: \texttt{webapp pushed:>=2025-01-01} + \texttt{topic:webapp}, języki Python/JS/TS/Java/Kotlin/C\#/Go/PHP/Ruby, 0--10 "gwiazdek"
        \item Z puli 100 repozytoriów wylosowano 25 do analizy
        \item Wstępna selekcja ręczna pod kątem możliwości budowy i uruchomienia
        \item Wybrane kryteria weryfikacji H1:
        \begin{itemize}
            \item end-to-end $\geq$ 60\%
            \item apply $|$ build $\geq$ 80\%, runtime $|$ apply $\geq$ 60\%
        \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}{Projekt H3 -- jakość konfiguracji}
    \begin{itemize}
        \item Cel: ocena jakości konfiguracji IaC za pomocą narzędzi analizy statycznej oraz wpływu \textit{prompt engineering} na jej poprawę
        \item Narzędzia: Hadolint (Dockerfile), Kube-linter (manifesty K8s)
        \item Wariant prompta z dobrymi praktykami IaC (m.in. non-root, przypinanie wersji, limity zasobów, sondy, securityContext)
        \item Zbiór: ten sam co w H1 (25 publicznych repozytoriów webowych z GitHuba)
        \item Wybrane kryteria weryfikacji:
        \begin{itemize}
            \item wariant bazowy: $\geq$ 60\% przebiegów z ostrzeżeniami
            \item wariant ulepszony: redukcja ostrzeżeń $\geq$ 20\% (mediana per repozytorium)
        \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}{Projekt H4 -- powtarzalność}
    \begin{itemize}
        \item Cel: ocena stabilności wyników przy parametrach sugerujących deterministyczność
        \item Zbiór: jak w H2 (POC1--POC5 o rosnącej złożoności)
        \item Modele: Gemini 2.5 Flash, DeepSeek V3.2 (GPT 5 mini nie wspiera \texttt{temperature=0})
        \item Ustawienia: 5 powtórzeń na kombinację, \texttt{temperature=0}, stały \texttt{seed}; bez apply
        \item Metryki: \texttt{diff\_ratio} między przebiegami oraz zmienność liczby kroków narzędziowych
        \item \texttt{diff\_ratio}: udział zmienionych linii wygenerowanych konfiguracji między dwoma przebiegami
    \end{itemize}
    \[
        \texttt{diff\_ratio} = \frac{\text{liczba linii zmienionych}}{\tfrac{1}{2}(\text{liczba linii w A} + \text{liczba linii w B})}
    \]
    \begin{itemize}
        \item Kryteria weryfikacji (wybrane):
        \begin{itemize}
            \item średni \texttt{diff\_ratio} $> 0{,}4$ dla każdej kombinacji
            \item odchylenie std. liczby kroków narzędziowych $> 20\%$ średniej
        \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}{Wnioski}
    \begin{itemize}
        \item Agenty LLM skutecznie generują działające IaC w prostych repozytoriach, ale skuteczność gwałtownie spada przy wzroście złożoności systemu
        \item Jakość generowanych konfiguracji rośnie po zastosowaniu odpowiednich wytycznych w prompcie systemowym (\textit{prompt engineering})
        \item Walidacja end-to-end (build/apply/runtime) jest kluczowa -- najwięcej błędów ujawnia się dopiero podczas uruchomienia, a nie przy sprawdzaniu składni
        \item Produkcyjnie potrzebne są mechanizmy oceny bezpieczeństwa konfiguracji, aby ograniczać podatność na ataki kontekstowe
    \end{itemize}
\end{frame}

\begin{frame}{Zagrożenia ważności (Threats to validity)}
    \begin{itemize}
        \item Zbiór H1: 25 relatywnie prostych aplikacji może nie uogólniać się na złożone systemy
        \item Walidacja runtime pojedynczym zapytaniem \texttt{curl} nie weryfikuje w pełni dostępności usługi
        \item Ograniczona liczba powtórzeń i brak większych (droższych) modeli może nie dawać pełnej perspektywy
        \item Brak porównania z konfiguracjami tworzonymi przez człowieka utrudnia ocenę absolutnej jakości
        \item Środowisko MicroK8s/lokalny rejestr obrazów może różnić się od produkcyjnych klastrów
        \item Agent po sklonowaniu usuwa gotowe konfiguracje IaC, aby nie ułatwiać zadania, co zmienia kontekst repozytorium
    \end{itemize}
\end{frame}

\begin{frame}{Dalsze kierunki}
    \begin{itemize}
        \item Rozszerzenie walidacji runtime o scenariusze integracyjne i testy zależności między usługami
        \item Automatyczna ocena "bezpieczeństwa" IaC (reguły semantyczne, zgodność z kodem źródłowym)
        \item Stabilizacja wyników: standaryzacja kontekstu, wielokrotne generacje i agregacja
        \item Pełna pętla sprzężenia zwrotnego: agent reaguje na błędy uruchomienia (np. brak zmiennej środowiskowej w logach)
        \item Wykorzystanie w systemie PaaS do produkcyjnego wdrażania aplikacji na klaster K8s po podaniu linku do repozytorium (\textbf{pierwotny cel pracy})
        \item Integracja z procesami CI/CD i udziałem człowieka w pętli
    \end{itemize}
\end{frame}
